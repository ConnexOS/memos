**Tech Stack:** Python 3.12 | ChromaDB (PersistentClient) | FastMCP | FastAPI + Jinja2 | 原生 JavaScript (Dashboard v2) | rank_bm25 | bge-large-zh-v1.5

```python
@mcp.tool()
def remember(text: str, metadata: dict = None) -> str:
    """标记待关注，写入 ChromaDB type=watchlist，供 L5 待关注面板展示。

    可通过 L5 Dashboard 将待关注内容转为知识/忽略/备注。
    """
    if len(text) > MAX_INPUT_LENGTH:
        return f"文本过长（{len(text)} 字符，上限 {MAX_INPUT_LENGTH}），请精简后重试。"

    watchlist_meta = {
        "type": "watchlist",
        "user_intent": "待关注",
        "created_at": time.time(),
        "processed": False,
        "processed_at": None,
        "project_id": _get_project_id(),
        "project_name": _get_project_name(_get_project_id()),
        "source": "remember",
    }
    if metadata:
        # 合并用户 metadata，但保留 watchlist 系统字段
        for k, v in metadata.items():
            if k not in ("type", "created_at", "processed", "user_intent"):
                watchlist_meta[k] = v

    mem = _get_memory()
    mid = mem.remember(text, metadata=watchlist_meta)
    if mid:
        return json.dumps({"id": mid, "status": "watchlist"}, ensure_ascii=False)
    return "保存失败"
```

```python
# src/memos/features/activity_log.py
"""活动日志采集 —— 三类事件（recall/写入/注入）的记录与轮转。

按天轮转，保留 30 天。非阻塞写入（异步文件追加）。
"""

import json
import logging
import os
import time
from datetime import datetime, timezone
from pathlib import Path

from ..config import config

logger = logging.getLogger(__name__)


def _get_log_path() -> Path:
    """获取活动日志目录路径。"""
    from ..config.models import get_memos_home

    custom = config.activity_log.log_path
    if custom:
        return Path(custom)
    return get_memos_home() / "etc"


def _get_log_filename(log_date: str = None) -> str:
    """获取当前日期对应的活动日志文件名。"""
    if log_date is None:
        log_date = datetime.now(timezone.utc).strftime("%Y-%m-%d")
    return f"activity_log_{log_date}.jsonl"


def _get_log_filepath(log_date: str = None) -> Path:
    return _get_log_path() / _get_log_filename(log_date)


def _cleanup_expired():
    """清理超 30 天的活动日志文件。每次首次写入时触发。"""
    retention = config.activity_log.retention_days
    log_dir = _get_log_path()
    if not log_dir.exists():
        return
    now = time.time()
    cutoff = now - retention * 86400
    for f in log_dir.glob("activity_log_*.jsonl"):
        try:
            if f.stat().st_mtime < cutoff:
                f.unlink()
                logger.info("已删除过期活动日志: %s", f.name)
        except OSError as e:
            logger.warning("删除过期日志失败 %s: %s", f.name, e)


def _append_event(event_data: dict):
    """追加一条事件到当前日期的活动日志文件。线程安全异步写入。"""
    _cleanup_expired()
    filepath = _get_log_filepath()
    filepath.parent.mkdir(parents=True, exist_ok=True)
    try:
        with open(filepath, "a", encoding="utf-8") as f:
            f.write(json.dumps(event_data, ensure_ascii=False) + "\n")
    except OSError as e:
        logger.error("写入活动日志失败: %s", e)


def log_recall(query: str, result_count: int, match_types: list[str], extra: dict = None):
    """记录 recall 调用事件。"""
    event = {
        "event": "recall",
        "timestamp": time.time(),
        "query": query[:200],
        "result_count": result_count,
        "match_types": match_types or [],
    }
    if extra:

def read_events(date: str = None, page: int = 1, page_size: int = 20) -> dict:
    """从指定日期的活动日志文件中读取事件（分页）。"""
    filepath = _get_log_filepath(date)
    if not filepath.exists():
        return {"items": [], "total": 0, "page": page, "page_size": page_size}

    events = []
    try:
        with open(filepath, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if line:
                    try:
                        events.append(json.loads(line))
                    except json.JSONDecodeError:
                        continue
    except OSError:
        return {"items": [], "total": 0, "page": page, "page_size": page_size}

    # 时间倒序
    events.sort(key=lambda x: x.get("timestamp", 0), reverse=True)
    total = len(events)
    start = (page - 1) * page_size
    end = start + page_size

    return {
        "items": events[start:end],
        "total": total,
        "page": page,
        "page_size": page_size,
    }
```

```python
# src/memos/server/task_handler.py
"""Stop Hook task 异步处理队列。

接收 TASK_EVAL → 格式校验 → 推入待处理队列 → 立即返回 200 OK
后台守护线程异步消费 → MEMOS LLM 结构化 → 写入 ChromaDB
"""

import json
import logging
import os
import queue
import threading
import time
from pathlib import Path

logger = logging.getLogger(__name__)


class TaskEvalQueue:
    """TASK_EVAL 待处理队列（内存队列 + 临时文件持久化防丢失）。"""

    def __init__(self, memory_instance=None, llm_caller=None):
        self._queue = queue.Queue()
        self._memory = memory_instance
        self._llm_caller = llm_caller
        self._running = False
        self._thread = None

    def start(self):
        """启动后台消费线程。"""
        if self._running:
            return
        self._running = True
        self._thread = threading.Thread(target=self._consume_loop, daemon=True, name="task-eval-consumer")
        self._thread.start()
        logger.info("TaskEvalQueue 后台消费线程已启动")

    def stop(self):
        """停止后台消费线程。"""
        self._running = False
        logger.info("TaskEvalQueue 已停止")

    def enqueue(self, task_eval: dict, session_id: str, project_id: str) -> bool:
        """推入一条任务到待处理队列。返回 True 表示入队成功。"""
        item = {
            "task_eval": task_eval,
            "session_id": session_id,
            "project_id": project_id,
            "received_at": time.time(),
        }
        self._queue.put_nowait(item)
        logger.info("TASK_EVAL 已入队: session=%s", session_id)
        return True

    def _consume_loop(self):
        """后台消费循环。"""
        while self._running:
            try:
                item = self._queue.get(timeout=1)
                self._process_item(item)
            except queue.Empty:
                continue
            except Exception as e:
                logger.error("task 消费异常: %s", e)

    def _structurize_task(self, task_eval: dict) -> dict | None:
        """调用 MEMOS LLM 将自评文本结构化为 task 对象。

        若 LLM 不可用，直接使用原始 task_eval 数据作为降级方案。
        """
        if self._llm_caller is None:
            return self._build_task_from_raw(task_eval)

        try:
            result = self._llm_caller(task_eval)
            if result:
                return json.loads(result) if isinstance(result, str) else result
        except Exception as e:
            logger.warning("MEMOS LLM 结构化失败，降级为直接使用原始数据: %s", e)

        return self._build_task_from_raw(task_eval)

    def _build_task_from_raw(self, task_eval: dict) -> dict:
        """直接使用原始 task_eval 数据构建 task（降级路径）。"""
        return {
            "project": task_eval.get("project", "general"),
            "goal": task_eval.get("goal", ""),
            "progress": {
                "done": task_eval.get("done", []),
                "todo": task_eval.get("todo", []),
                "blocked": task_eval.get("blocked", []),
            },
            "next_steps": task_eval.get("next_steps", task_eval.get("todo", [])),
            "confidence": 0.7,
        }

    def _process_item(self, item: dict):
        """处理单条 TASK_EVAL 项。"""
        task_eval = item["task_eval"]
        session_id = item["session_id"]
        project_id = item["project_id"]

        # 1. 结构化
        structured = self._structurize_task(task_eval)
        if structured is None:
            logger.warning("task 结构化失败，跳过: session=%s", session_id)
            return

        # 2. 写入 ChromaDB
        if self._memory is None:
            logger.warning("memory 实例不可用，无法写入 task")
            return

        now = time.time()
        metadata = {
            "type": "task",
            "source": "stop_hook",
            "source_info": {
                "session_id": session_id,
                "conversation_count": 0,
                "note": "",
            },
            "project_id": project_id,
            "confidence": structured.get("confidence", 0.7),
            "archived": False,
            "paused": False,
            "previous_versions": json.dumps([]),
            "created_at": now,
            "updated_at": now,
        }

        # 同链检查：相同 project 的现有 active task
        try:
            existing = self._memory.recall(
                query=structured.get("project", ""),
                top_k=1,
                where={"type": "task"},
                project_id=project_id,
            )
            if existing and len(existing) > 0:
                # 同 project → 更新（保留 previous_versions）
                old = existing[0]
                old_meta = old.get("metadata", {})
                if old_meta.get("project") == structured.get("project"):
                    old_id = old.get("id")
                    if old_id:
                        prev = json.loads(old_meta.get("previous_versions", "[]"))
                        prev.append({
                            "project": old_meta.get("project", ""),
                            "goal": old_meta.get("goal", ""),
                            "progress": old_meta.get("progress", {}),
                            "updated_at": old_meta.get("updated_at", now),
                        })
                        metadata["previous_versions"] = json.dumps(prev)
                        self._memory.update_memory(old_id, new_metadata=metadata)
                        logger.info("task 已更新（同链覆盖）: project=%s", structured.get("project", "general"))
                        return
        except Exception as e:
            logger.warning("task 同链检查失败，尝试新建: %s", e)

        # 3. 新建 task
        content_text = json.dumps(structured, ensure_ascii=False)
        self._memory.remember(content_text, metadata=metadata)
        logger.info("task 已创建: project=%s", structured.get("project", "general"))

        # 4. 冷启动标记
        cold_start_file = Path(os.environ.get("CLAUDE_PROJECT_DIR", os.getcwd())) / "etc" / ".cold_start_done"
        if not cold_start_file.exists():
            cold_start_file.parent.mkdir(parents=True, exist_ok=True)
            cold_start_file.write_text(
                json.dumps({"done": True, "created_at": now}, ensure_ascii=False),
                encoding="utf-8",
            )
            logger.info("冷启动标记已创建")
```

```python
def _inject_active_task(mem, pid: str) -> str:
    """检索最近的 active task 并构造注入文本。

    选择策略：latest 1 条未归档 + 未暂停（archived=false, paused=false）。
    注入优先级：task > briefing（task 信息在 additionalContext 中最先出现）。
    """
    if mem is None:
        return ""

    try:
        # 按 updated_at 倒序取最新的 active task
        results = mem.list_memories(
            project_id=pid,
            type_filter="task",
            limit=1,
        )
        if not results:
            return ""

        task = results[0]
        meta = task.get("metadata", {})
        if meta.get("archived", False) or meta.get("paused", False):
            return ""

        # 解析 content
        content_text = task.get("document", "")
        try:
            content = json.loads(content_text) if content_text else {}
        except (json.JSONDecodeError, TypeError):
            content = {}

        project_name = content.get("project", meta.get("source_info", {}).get("project", ""))
        goal = content.get("goal", "")
        progress = content.get("progress", {})
        done = progress.get("done", [])
        todo = progress.get("todo", [])
        blocked = progress.get("blocked", [])

        lines = ["[当前任务]"]
        if project_name:
            lines.append(f"项目: {project_name}")
        if goal:
            lines.append(f"目标: {goal}")
        if done:
            lines.append(f"已完成: {'; '.join(done)}")
        if todo:
            lines.append(f"待完成: {'; '.join(todo)}")
        if blocked:
            lines.append(f"阻塞项: {'; '.join(blocked)}")

        return "\n".join(lines)

    except Exception as e:
        logger.warning("task 注入失败: %s", e)
        return ""
```

# src/memos/features/briefing.py
"""简报生成 —— 主路径（MEMOS LLM）+ 兜底路径（纯模板拼接）。"""

import json
import logging
import time
from datetime import datetime
from typing import Any

logger = logging.getLogger(__name__)


def build_fallback_briefing(today_task: dict = None, today_events: list = None, event_count: int = 0) -> dict:
    """构建兜底简报（quality=simple，不调 LLM）。

    参数：
        today_task: 当日最新 task 的 content（dict，含 project/done/todo/blocked/next_steps）
        today_events: 当日活动日志前 5 条
        event_count: 当日事件总数

    返回简报 dict。
    """
    task = today_task or {}
    events = (today_events or [])[:5]
    done = task.get("progress", {}).get("done", [])
    todo = task.get("progress", {}).get("todo", [])
    blocked = task.get("progress", {}).get("blocked", [])
    next_steps = task.get("next_steps", [])

    # summary
    if task:
        total = len(done) + len(todo)
        summary = f"今日共 {event_count} 轮对话，当前任务进度：{len(done)}/{total}"
    else:
        summary = "今日无活跃任务记录"

    # task_status
    if task:
        project = task.get("project", "")
        task_status = (
            f"项目: {project}\n"
            f"已完成: {done}\n"
            f"待完成: {todo}\n"
            f"阻塞: {blocked}"
        )
    else:
        task_status = "无"

    # key_events
    key_events = []
    for e in events:
        ts = e.get("timestamp", 0)
        time_str = time.strftime("%H:%M", time.localtime(ts)) if ts else ""
        evt = e.get("event", "")
        detail = e.get("query", e.get("type", ""))
        key_events.append(f"{time_str}: {evt}: {detail}")

    # plan_tomorrow
    if task and next_steps:
        plan_tomorrow = "根据当前进度，下一步计划：" + "；".join(next_steps)
    else:
        plan_tomorrow = "无"

    return {
        "summary": summary,
        "task_status": task_status,
        "key_events": key_events,
        "new_knowledge": [],
        "plan_tomorrow": plan_tomorrow,
        "quality": "simple",
        "source": "fallback_hook",
    }


def build_full_briefing(task_data: dict, activity_log: list, new_knowledge: list, llm_caller) -> dict | None:
    """调用 MEMOS LLM 生成完整简报（quality=full）。

    参数：
        task_data: 当前 task 数据
        activity_log: 当日活动日志
        new_knowledge: 当日新 L3 知识
        llm_caller: 可调用对象，接收 prompt 文本返回 JSON 字符串

    返回结构化 dict，或 None（LLM 调用失败）。
    """
    prompt_input = json.dumps({
        "task": task_data,
        "activity_log": activity_log[:50],
        "new_knowledge": new_knowledge,
    }, ensure_ascii=False)

    system_prompt = (
        "你是一个项目简报生成引擎。根据输入的当日数据，生成结构化项目简报。\n\n"
        "输出 JSON 格式如下，不含额外说明：\n"
        '{"summary": "今日工作的 1-2 句话概述",\n'
        ' "task_status": "当前任务进度的一句话摘要",\n'
        ' "key_events": ["按时间顺序的关键事件, 每条 1-2 句话"],\n'
        ' "new_knowledge": ["当日新增的知识条目摘要, 每条一句话"],\n'
        ' "plan_tomorrow": "明日计划的 1-2 句话"}\n\n'
        "约束：\n"
        "- 如果当日没有任何 task/事件/知识，对应的字段输出空数组\n"
        "- 如果当日仅有 0-5 轮对话，summary 注明'今日对话较少'\n"
        "- 输出必须是合法 JSON，不得包含 ```json 标记"
    )

    user_prompt = f"请根据以下当日数据生成今日项目简报：\n\n{prompt_input}"

    try:
        result = llm_caller(system_prompt, user_prompt)
        if result:
            parsed = json.loads(result) if isinstance(result, str) else result
            parsed["quality"] = "full"
            parsed["source"] = "scheduler"
            return parsed
    except Exception as e:
        logger.error("完整简报生成失败: %s", e)

    return None

```python
# src/memos/features/scheduler.py
"""Dashboard 调度线程 —— 定时生成简报。"""

import logging
import threading
import time
from datetime import datetime

logger = logging.getLogger(__name__)


class SchedulerThread:
    """简报调度线程。在配置时间（默认 23:00）触发生成。"""

    def __init__(self, briefing_generator=None, memory_instance=None):
        self._running = False
        self._thread = None
        self._generator = briefing_generator
        self._memory = memory_instance

    def start(self):
        if self._running:
            return
        self._running = True
        self._thread = threading.Thread(target=self._run_loop, daemon=True, name="scheduler")
        self._thread.start()
        logger.info("SchedulerThread 已启动")

    def stop(self):
        self._running = False
        logger.info("SchedulerThread 已停止")

    def _run_loop(self):
        """每分钟检查一次是否需要生成简报。"""
        while self._running:
            try:
                self._check_and_generate()
            except Exception as e:
                logger.error("调度循环异常: %s", e)
            time.sleep(60)

    def _check_and_generate(self):
        """检查当前时间是否到达配置的生成时间，且今日尚未生成。"""
        from ..config import config as _cfg

        now = datetime.now()
        generate_time_str = _cfg.scheduler_briefing.generate_time  # "23:00"
        try:
            gen_hour, gen_min = map(int, generate_time_str.split(":"))
        except (ValueError, AttributeError):
            logger.warning("简报生成时间配置无效: %s", generate_time_str)
            return

        if now.hour != gen_hour or now.minute != gen_min:
            return

        # 检查今日是否已有简报
        today = now.strftime("%Y-%m-%d")
        if self._has_today_briefing(today):
            logger.debug("今日(%s)已有简报，跳过", today)
            return

        # 触发主路径生成
        if self._generator:
            self._generator()
            logger.info("调度器: 简报已生成 (%s)", today)

    def _has_today_briefing(self, today: str) -> bool:
        """检查今日是否已有 briefing 记录（含兜底和调度器生成）。"""
        if self._memory is None:
            return False
        try:
            results = self._memory.list_memories(
                type_filter="briefing",
                limit=1,
            )
            for item in results:
                meta = item.get("metadata", {})
                if meta.get("briefing_date") == today:
                    return True
            return False
        except Exception as e:
            logger.warning("检查今日简报失败: %s", e)
            return False
```

```python
        result = read_events(date=d, page=1, page_size=100)
        all_items.extend(result.get("items", []))

    # 时间倒序
    all_items.sort(key=lambda x: x.get("timestamp", 0), reverse=True)

    total = len(all_items)
    start = (page - 1) * page_size
    end = start + page_size

    return JSONResponse({
        "items": all_items[start:end],
        "total": total,
        "page": page,
        "page_size": page_size,
    })
```

```python
    request: Request,

    archived_ids = []
    for item in results:

        created_at = meta.get("created_at", 0)
        if created_at > 0 and now - created_at > 30 * 86400 and not meta.get("archived", False):
            meta["archived"] = True
            mem.update_memory(item["id"], new_metadata={"archived": True})
            archived_ids.append(item["id"])

    return JSONResponse({
        "items": results,
        "total": len(results),
        "page": page,
        "page_size": page_size,
        "archived_count": len(archived_ids),
    })


@router.post("/api/v2/watchlist/{memory_id}/to-knowledge")
async def watchlist_to_knowledge(memory_id: str, request: Request):
    """将待关注转为知识。"""
    body = await request.json()
    target_type = body.get("type")  # solution/decision/lesson/process
    preview_edit = body.get("preview_edit", None)

    if target_type not in ("solution", "decision", "lesson", "process"):
        return JSONResponse({"error": "无效类型"}, status_code=400)

    mem = _get_memory()
    existing = mem.get_memory(memory_id)
    if existing is None:
        return JSONResponse({"error": "记忆不存在"}, status_code=404)

    text = preview_edit or existing.get("document", "")

    # 写入为目标类型
    new_meta = {
        "type": target_type,
        "source": "watchlist_conversion",
        "project_id": existing.get("metadata", {}).get("project_id", ""),
        "timestamp": time.time(),
    }
    new_id = mem.remember(text, metadata=new_meta)

    # 原 watchlist 标记 processed
    mem.update_memory(memory_id, new_metadata={"processed": True, "processed_at": time.time()})

    return JSONResponse({"id": new_id, "status": "converted"})
```

```python
    request: Request,
    days: int = Query(default=7, ge=1, le=90),
    confirmed: bool = None,
    type: str = None,
    page: int = Query(default=1, ge=1),
    page_size: int = Query(default=20, ge=1, le=100),
):

    # 查询 L2+L3 类型（task/briefing/solution/decision/lesson/process）
    results = mem.list_memories(
        type_filter=type or None,
        limit=page_size,
        offset=(page - 1) * page_size,
    )

    # 过滤：近 N 天且 confirmed != true
    now = time.time()
    cutoff = now - days * 86400
    filtered = []
    for item in results:
        meta = item.get("metadata", {})
        ts = meta.get("timestamp", 0)
        if ts < cutoff:
            continue
        if confirmed is False and meta.get("confirmed", False):
            continue
        if confirmed is True and not meta.get("confirmed", False):
            continue
        filtered.append(item)

    return JSONResponse({
        "items": filtered,
        "total": len(filtered),
        "page": page,
        "page_size": page_size,
    })
```

```python
def record_baseline(metrics: dict):
    """记录一条性能基线数据。

    metrics 结构：
    {
        "timestamp": 1747000000,
        "prompt_hook_p50": 0.05,
        "prompt_hook_p95": 0.10,

        try:
            records = json.loads(path.read_text(encoding="utf-8"))
        except (json.JSONDecodeError, OSError):
            records = []

    metrics["timestamp"] = time.time()
    records.append(metrics)

    # 保留最近 10 条
    if len(records) > _MAX_RECORDS:
        records = records[-_MAX_RECORDS:]

    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(json.dumps(records, ensure_ascii=False, indent=2), encoding="utf-8")
```
